55 {
56 (*this)["algorithm"] = FLANN_INDEX_AUTOTUNED;

229 void evaluate_kmeans(CostData& cost)
230 {
231 StartStopTimer t;
232 int checks;
233 const int nn = 1;
234
235 Logger::info("KMeansTree using params: max_iterations=%d, branching=%d\n",
236 get_param<int>(cost.params,"iterations"),
237 get_param<int>(cost.params,"branching"));
238 KMeansIndex<Distance> kmeans(sampledDataset_, cost.params, distance_);
239 // measure index build time
240 t.start();
241 kmeans.buildIndex();
242 t.stop();
243 float buildTime = (float)t.value;
244
245 // measure search time
246 float searchTime = test_index_precision(kmeans, sampledDataset_, testDataset_, gt_matches_, target_precision_, checks, distance_, nn);
247
248 float datasetMemory = float(sampledDataset_.rows * sampledDataset_.cols * sizeof(float));
249 cost.memoryCost = (kmeans.usedMemory() + datasetMemory) / datasetMemory;
250 cost.searchTimeCost = searchTime;
251 cost.buildTimeCost = buildTime;
252 Logger::info("KMeansTree buildTime=%g, searchTime=%g, build_weight=%g\n", buildTime, searchTime, build_weight_);
253 }

419 IndexParams estimateBuildParams()
420 {
421 std::vector<CostData> costs;
422
423 int sampleSize = int(sample_fraction_ * dataset_.rows);
424 int testSampleSize = std::min(sampleSize / 10, 1000);
425
426 Logger::info("Entering autotuning, dataset size: %d, sampleSize: %d, testSampleSize: %d, target precision: %g\n", dataset_.rows, sampleSize, testSampleSize, target_precision_);
427
428 // For a very small dataset, it makes no sense to build any fancy index, just
429 // use linear search
430 if (testSampleSize < 10) {
431 Logger::info("Choosing linear, dataset too small\n");
432 return LinearIndexParams();
433 }
434
435 // We use a fraction of the original dataset to speedup the autotune algorithm
436 sampledDataset_ = random_sample(dataset_, sampleSize);
437 // We use a cross-validation approach, first we sample a testset from the dataset
438 testDataset_ = random_sample(sampledDataset_, testSampleSize, true);
439
440 // We compute the ground truth using linear search
441 Logger::info("Computing ground truth... \n");
442 gt_matches_ = Matrix<int>(new int[testDataset_.rows], testDataset_.rows, 1);
443 StartStopTimer t;
444 t.start();
445 compute_ground_truth<Distance>(sampledDataset_, testDataset_, gt_matches_, 0, distance_);
446 t.stop();
447
448 CostData linear_cost;
449 linear_cost.searchTimeCost = (float)t.value;
450 linear_cost.buildTimeCost = 0;
451 linear_cost.memoryCost = 0;
452 linear_cost.params["algorithm"] = FLANN_INDEX_LINEAR;
453
454 costs.push_back(linear_cost);
455
456 // Start parameter autotune process
457 Logger::info("Autotuning parameters...\n");
458
459 optimizeKMeans(costs);
460 optimizeKDTree(costs);
461
462 float bestTimeCost = costs[0].searchTimeCost;
463 for (size_t i = 0; i < costs.size(); ++i) {
464 float timeCost = costs[i].buildTimeCost * build_weight_ + costs[i].searchTimeCost;
465 if (timeCost < bestTimeCost) {
466 bestTimeCost = timeCost;
467 }
468 }
469
470 float bestCost = costs[0].searchTimeCost / bestTimeCost;
471 IndexParams bestParams = costs[0].params;
472 if (bestTimeCost > 0) {
473 for (size_t i = 0; i < costs.size(); ++i) {
474 float crtCost = (costs[i].buildTimeCost * build_weight_ + costs[i].searchTimeCost) / bestTimeCost +
475 memory_weight_ * costs[i].memoryCost;
476 if (crtCost < bestCost) {
477 bestCost = crtCost;
478 bestParams = costs[i].params;
479 }
480 }
481 }
482
483 delete[] gt_matches_.data;
484 delete[] testDataset_.data;
485 delete[] sampledDataset_.data;
486
487 return bestParams;
488 }

497 float estimateSearchParams(SearchParams& searchParams)
498 {
499 const int nn = 1;
500 const size_t SAMPLE_COUNT = 1000;
501
502 assert(bestIndex_ != NULL); // must have a valid index
503
504 float speedup = 0;
505
506 int samples = (int)std::min(dataset_.rows / 10, SAMPLE_COUNT);
507 if (samples > 0) {
508 Matrix<ElementType> testDataset = random_sample(dataset_, samples);
509
510 Logger::info("Computing ground truth\n");
511
512 // we need to compute the ground truth first
513 Matrix<int> gt_matches(new int[testDataset.rows], testDataset.rows, 1);
514 StartStopTimer t;
515 t.start();
516 compute_ground_truth<Distance>(dataset_, testDataset, gt_matches, 1, distance_);
517 t.stop();
518 float linear = (float)t.value;
519
520 int checks;
521 Logger::info("Estimating number of checks\n");
522
523 float searchTime;
524 float cb_index;
525 if (bestIndex_->getType() == FLANN_INDEX_KMEANS) {
526 Logger::info("KMeans algorithm, estimating cluster border factor\n");
527 KMeansIndex<Distance>* kmeans = (KMeansIndex<Distance>*)bestIndex_;
528 float bestSearchTime = -1;
529 float best_cb_index = -1;
530 int best_checks = -1;
531 for (cb_index = 0; cb_index < 1.1f; cb_index += 0.2f) {
532 kmeans->set_cb_index(cb_index);
533 searchTime = test_index_precision(*kmeans, dataset_, testDataset, gt_matches, target_precision_, checks, distance_, nn, 1);
534 if ((searchTime < bestSearchTime) || (bestSearchTime == -1)) {
535 bestSearchTime = searchTime;
536 best_cb_index = cb_index;
537 best_checks = checks;
538 }
539 }
540 searchTime = bestSearchTime;
541 cb_index = best_cb_index;
542 checks = best_checks;
543
544 kmeans->set_cb_index(best_cb_index);
545 Logger::info("Optimum cb_index: %g\n", cb_index);
546 bestParams_["cb_index"] = cb_index;
547 }
548 else {
549 searchTime = test_index_precision(*bestIndex_, dataset_, testDataset, gt_matches, target_precision_, checks, distance_, nn, 1);
550 }
551
552 Logger::info("Required number of checks: %d \n", checks);
553 searchParams["checks"] = checks;
554
555 speedup = linear / searchTime;
556
557 delete[] gt_matches.data;
558 delete[] testDataset.data;
559 }
560
561 return speedup;
562 }
