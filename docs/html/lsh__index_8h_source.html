112 void addIndex(const Matrix<ElementType>& wholeData, const Matrix<ElementType>& additionalData)
113 {
114 tables_.resize(table_number_);
115 for (unsigned int i = 0; i < table_number_; ++i) {
116 lsh::LshTable<ElementType>& table = tables_[i];
117 // Add the features to the table with indexed offset
118 table.add((int)(wholeData.rows - additionalData.rows), additionalData);
119 }
120 dataset_ = wholeData;
121 }

126 void buildIndex()
127 {
128 std::vector<size_t> indices(feature_size_ * CHAR_BIT);
129
130 tables_.resize(table_number_);
131 for (unsigned int i = 0; i < table_number_; ++i) {
132
133 //re-initialize the random indices table that the LshTable will use to pick its sub-dimensions
134 if( (indices.size() == feature_size_ * CHAR_BIT) || (indices.size() < key_size_) )
135 {
136 indices.resize( feature_size_ * CHAR_BIT );
137 for (size_t j = 0; j < feature_size_ * CHAR_BIT; ++j)
138 indices[j] = j;
139 std::random_shuffle(indices.begin(), indices.end());
140 }
141
142 lsh::LshTable<ElementType>& table = tables_[i];
143 table = lsh::LshTable<ElementType>(feature_size_, key_size_, indices);
144
145 // Add the features to the table with offset 0
146 table.add(0, dataset_);
147 }
148 }

218 virtual void knnSearch(const Matrix<ElementType>& queries, Matrix<int>& indices, Matrix<DistanceType>& dists, int knn, const SearchParams& params)
219 {
220 assert(queries.cols == veclen());
221 assert(indices.rows >= queries.rows);
222 assert(dists.rows >= queries.rows);
223 assert(int(indices.cols) >= knn);
224 assert(int(dists.cols) >= knn);
225
226
227 KNNUniqueResultSet<DistanceType> resultSet(knn);
228 for (size_t i = 0; i < queries.rows; i++) {
229 resultSet.clear();
230 std::fill_n(indices[i], knn, -1);
231 std::fill_n(dists[i], knn, std::numeric_limits<DistanceType>::max());
232 findNeighbors(resultSet, queries[i], params);
233 if (get_param(params,"sorted",true)) resultSet.sortAndCopy(indices[i], dists[i], knn);
234 else resultSet.copy(indices[i], dists[i], knn);
235 }
236 }

271 void fill_xor_mask(lsh::BucketKey key, int lowest_index, unsigned int level,
272 std::vector<lsh::BucketKey>& xor_masks)
273 {
274 xor_masks.push_back(key);
275 if (level == 0) return;
276 for (int index = lowest_index - 1; index >= 0; --index) {
277 // Create a new key
278 lsh::BucketKey new_key = key | (1 << index);
279 fill_xor_mask(new_key, index, level - 1, xor_masks);
280 }

291 void getNeighbors(const ElementType* vec, bool /*do_radius*/, float radius, bool do_k, unsigned int k_nn,
292 float& /*checked_average*/)
293 {
294 static std::vector<ScoreIndexPair> score_index_heap;
295
296 if (do_k) {
297 unsigned int worst_score = std::numeric_limits<unsigned int>::max();
298 typename std::vector<lsh::LshTable<ElementType> >::const_iterator table = tables_.begin();
299 typename std::vector<lsh::LshTable<ElementType> >::const_iterator table_end = tables_.end();
300 for (; table != table_end; ++table) {
301 size_t key = table->getKey(vec);
302 std::vector<lsh::BucketKey>::const_iterator xor_mask = xor_masks_.begin();
303 std::vector<lsh::BucketKey>::const_iterator xor_mask_end = xor_masks_.end();
304 for (; xor_mask != xor_mask_end; ++xor_mask) {
305 size_t sub_key = key ^ (*xor_mask);
306 const lsh::Bucket* bucket = table->getBucketFromKey(sub_key);
307 if (bucket == 0) continue;
308
309 // Go over each descriptor index
310 std::vector<lsh::FeatureIndex>::const_iterator training_index = bucket->begin();
311 std::vector<lsh::FeatureIndex>::const_iterator last_training_index = bucket->end();
312 DistanceType hamming_distance;
313
314 // Process the rest of the candidates
315 for (; training_index < last_training_index; ++training_index) {
316 hamming_distance = distance_(vec, dataset_[*training_index], dataset_.cols);
317
318 if (hamming_distance < worst_score) {
319 // Insert the new element
320 score_index_heap.push_back(ScoreIndexPair(hamming_distance, training_index));
321 std::push_heap(score_index_heap.begin(), score_index_heap.end());
322
323 if (score_index_heap.size() > (unsigned int)k_nn) {
324 // Remove the highest distance value as we have too many elements
325 std::pop_heap(score_index_heap.begin(), score_index_heap.end());
326 score_index_heap.pop_back();
327 // Keep track of the worst score
328 worst_score = score_index_heap.front().first;
329 }
330 }
331 }
332 }
333 }
334 }
335 else {
336 typename std::vector<lsh::LshTable<ElementType> >::const_iterator table = tables_.begin();
337 typename std::vector<lsh::LshTable<ElementType> >::const_iterator table_end = tables_.end();
338 for (; table != table_end; ++table) {
339 size_t key = table->getKey(vec);
340 std::vector<lsh::BucketKey>::const_iterator xor_mask = xor_masks_.begin();
341 std::vector<lsh::BucketKey>::const_iterator xor_mask_end = xor_masks_.end();
342 for (; xor_mask != xor_mask_end; ++xor_mask) {
343 size_t sub_key = key ^ (*xor_mask);
344 const lsh::Bucket* bucket = table->getBucketFromKey(sub_key);
345 if (bucket == 0) continue;
346
347 // Go over each descriptor index
348 std::vector<lsh::FeatureIndex>::const_iterator training_index = bucket->begin();
349 std::vector<lsh::FeatureIndex>::const_iterator last_training_index = bucket->end();
350 DistanceType hamming_distance;
351
352 // Process the rest of the candidates
353 for (; training_index < last_training_index; ++training_index) {
354 // Compute the Hamming distance
355 hamming_distance = distance_(vec, dataset_[*training_index], dataset_.cols);
356 if (hamming_distance < radius) score_index_heap.push_back(ScoreIndexPair(hamming_distance, training_index));
357 }
358 }
359 }
360 }
361 }

367 void getNeighbors(const ElementType* vec, ResultSet<DistanceType>& result)
368 {
369 typename std::vector<lsh::LshTable<ElementType> >::const_iterator table = tables_.begin();
370 typename std::vector<lsh::LshTable<ElementType> >::const_iterator table_end = tables_.end();
371 for (; table != table_end; ++table) {
372 size_t key = table->getKey(vec);
373 std::vector<lsh::BucketKey>::const_iterator xor_mask = xor_masks_.begin();
374 std::vector<lsh::BucketKey>::const_iterator xor_mask_end = xor_masks_.end();
375 for (; xor_mask != xor_mask_end; ++xor_mask) {
376 size_t sub_key = key ^ (*xor_mask);
377 const lsh::Bucket* bucket = table->getBucketFromKey((lsh::BucketKey)sub_key);
378 if (bucket == 0) continue;
379
380 // Go over each descriptor index
381 std::vector<lsh::FeatureIndex>::const_iterator training_index = bucket->begin();
382 std::vector<lsh::FeatureIndex>::const_iterator last_training_index = bucket->end();
383 DistanceType hamming_distance;
384
385 // Process the rest of the candidates
386 for (; training_index < last_training_index; ++training_index) {
387 // Compute the Hamming distance
388 hamming_distance = distance_(vec, dataset_[*training_index], (int)dataset_.cols);
389 result.addPoint(hamming_distance, *training_index);
390 }
391 }
392 }
