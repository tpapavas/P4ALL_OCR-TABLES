293Index_<T>::Index_(const Mat& dataset, const ::cvflann::IndexParams& params)
294{
295 printf("[WARNING] The cv::flann::Index_<T> class is deperecated, use cv::flann::GenericIndex<Distance> instead\n");
296
297 CV_Assert(dataset.type() == CvType<ElementType>::type());
