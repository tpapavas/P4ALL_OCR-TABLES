310static inline unsigned int
311fz_irect_width(fz_irect r)
312{
313 unsigned int w;
314 if (r.x0 >= r.x1)
315 return 0;
316 /* Check for w overflowing. This should never happen, but
317 * if it does, it's pretty likely an indication of a severe
318 * problem. */
319 w = (unsigned int)r.x1 - r.x0;
320 assert((int)w >= 0);
321 if ((int)w < 0)
322 return 0;
323 return (int)w;
324}
325
329static inline int
330fz_irect_height(fz_irect r)
331{
332 unsigned int h;
333 if (r.y0 >= r.y1)
334 return 0;
335 /* Check for h overflowing. This should never happen, but
336 * if it does, it's pretty likely an indication of a severe
337 * problem. */
338 h = (unsigned int)(r.y1 - r.y0);
339 assert((int)h >= 0);
340 if ((int)h < 0)
341 return 0;
342 return (int)h;
343}
