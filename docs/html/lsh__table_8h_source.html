171 void add(unsigned int value, const ElementType* feature)
172 {
173 // Add the value to the corresponding bucket
174 BucketKey key = (lsh::BucketKey)getKey(feature);
175
176 switch (speed_level_) {
177 case kArray:
178 // That means we get the buckets from an array
179 buckets_speed_[key].push_back(value);
180 break;
181 case kBitsetHash:
182 // That means we can check the bitset for the presence of a key
183 key_bitset_.set(key);
184 buckets_space_[key].push_back(value);
185 break;
186 case kHash:
187 {
188 // That means we have to check for the hash table for the presence of a key
189 buckets_space_[key].push_back(value);
190 break;
191 }
192 }
193 }
194
199 void add(int indexed_ofst, Matrix<ElementType> dataset)
200 {
201#if USE_UNORDERED_MAP
202 buckets_space_.rehash((buckets_space_.size() + dataset.rows) * 1.2);
203#endif
204 // Add the features to the table
205 for (unsigned int i = 0; i < dataset.rows; ++i) add(i + indexed_ofst, dataset[i]);
206 // Now that the table is full, optimize it for speed/space
207 optimize();
208 }

214 inline const Bucket* getBucketFromKey(BucketKey key) const
215 {
216 // Generate other buckets
217 switch (speed_level_) {
218 case kArray:
219 // That means we get the buckets from an array
220 return &buckets_speed_[key];
221 break;
222 case kBitsetHash:
223 // That means we can check the bitset for the presence of a key
224 if (key_bitset_.test(key)) return &buckets_space_.find(key)->second;
225 else return 0;
226 break;
227 case kHash:
228 {
229 // That means we have to check for the hash table for the presence of a key
230 BucketsSpace::const_iterator bucket_it, bucket_end = buckets_space_.end();
231 bucket_it = buckets_space_.find(key);
232 // Stop here if that bucket does not exist
233 if (bucket_it == bucket_end) return 0;
234 else return &bucket_it->second;
235 break;
236 }
237 }
238 return 0;
239 }

261 enum SpeedLevel

268 void initialize(size_t key_size)
269 {
270 const size_t key_size_lower_bound = 1;
271 //a value (size_t(1) << key_size) must fit the size_t type so key_size has to be strictly less than size of size_t
272 const size_t key_size_upper_bound = std::min(sizeof(BucketKey) * CHAR_BIT + 1, sizeof(size_t) * CHAR_BIT);
273 if (key_size < key_size_lower_bound || key_size >= key_size_upper_bound)
274 {
275 std::stringstream errorMessage;
276 errorMessage << "Invalid key_size (=" << key_size << "). Valid values for your system are " << key_size_lower_bound << " <= key_size < " << key_size_upper_bound << ".";
277 CV_Error(CV_StsBadArg, errorMessage.str());
278 }
279
280 speed_level_ = kHash;
281 key_size_ = (unsigned)key_size;
282 }
283
286 void optimize()
287 {
288 // If we are already using the fast storage, no need to do anything
289 if (speed_level_ == kArray) return;
290
291 // Use an array if it will be more than half full
292 if (buckets_space_.size() > ((size_t(1) << key_size_) / 2)) {
293 speed_level_ = kArray;
294 // Fill the array version of it
295 buckets_speed_.resize(size_t(1) << key_size_);
296 for (BucketsSpace::const_iterator key_bucket = buckets_space_.begin(); key_bucket != buckets_space_.end(); ++key_bucket) buckets_speed_[key_bucket->first] = key_bucket->second;
297
298 // Empty the hash table
299 buckets_space_.clear();
300 return;
301 }
302
303 // If the bitset is going to use less than 10% of the RAM of the hash map (at least 1 size_t for the key and two
304 // for the vector) or less than 512MB (key_size_ <= 30)
305 if (((std::max(buckets_space_.size(), buckets_speed_.size()) * CHAR_BIT * 3 * sizeof(BucketKey)) / 10
306 >= (size_t(1) << key_size_)) || (key_size_ <= 32)) {
307 speed_level_ = kBitsetHash;
308 key_bitset_.resize(size_t(1) << key_size_);
309 key_bitset_.reset();
310 // Try with the BucketsSpace
311 for (BucketsSpace::const_iterator key_bucket = buckets_space_.begin(); key_bucket != buckets_space_.end(); ++key_bucket) key_bitset_.set(key_bucket->first);
312 }
313 else {
314 speed_level_ = kHash;
315 key_bitset_.clear();
316 }
317 }

351template<>
352inline LshTable<unsigned char>::LshTable( unsigned int feature_size,
353 unsigned int subsignature_size,
354 std::vector<size_t> & indices )
355{
356 feature_size_ = feature_size;
357 initialize(subsignature_size);
358 // Allocate the mask
359 mask_ = std::vector<size_t>((feature_size * sizeof(char) + sizeof(size_t) - 1) / sizeof(size_t), 0);
360
361 // Generate a random set of order of subsignature_size_ bits
362 for (unsigned int i = 0; i < key_size_; ++i) {
363 //Ensure the Nth bit will be selected only once among the different LshTables
364 //to avoid having two different tables with signatures sharing many dimensions/many bits
365 size_t index = indices[0];
366 indices.erase( indices.begin() );
367
368 // Set that bit in the mask
369 size_t divisor = CHAR_BIT * sizeof(size_t);
370 size_t idx = index / divisor; //pick the right size_t index
371 mask_[idx] |= size_t(1) << (index % divisor); //use modulo to find the bit offset
372 }
373
374 // Set to 1 if you want to display the mask for debug
375#if 0
376 {
377 size_t bcount = 0;
378 BOOST_FOREACH(size_t mask_block, mask_){
379 out << std::setw(sizeof(size_t) * CHAR_BIT / 4) << std::setfill('0') << std::hex << mask_block
380 << std::endl;
381 bcount += __builtin_popcountll(mask_block);
382 }
383 out << "bit count : " << std::dec << bcount << std::endl;
384 out << "mask size : " << mask_.size() << std::endl;

393template<>
394inline size_t LshTable<unsigned char>::getKey(const unsigned char* feature) const
395{
396 // no need to check if T is dividable by sizeof(size_t) like in the Hamming
397 // distance computation as we have a mask
398 // FIXIT: This is bad assumption, because we reading tail bytes after of the allocated features buffer
399 const size_t* feature_block_ptr = reinterpret_cast<const size_t*> ((const void*)feature);
400
401 // Figure out the subsignature of the feature
402 // Given the feature ABCDEF, and the mask 001011, the output will be
403 // 000CEF
404 size_t subsignature = 0;
405 size_t bit_index = 1;
406
407 for (unsigned i = 0; i < feature_size_; i += sizeof(size_t)) {
408 // get the mask and signature blocks
409 size_t feature_block;
410 if (i <= feature_size_ - sizeof(size_t))
411 {
412 feature_block = *feature_block_ptr;
413 }
414 else
415 {
416 size_t tmp = 0;
417 memcpy(&tmp, feature_block_ptr, feature_size_ - i); // preserve bytes order
418 feature_block = tmp;
419 }
420 size_t mask_block = mask_[i / sizeof(size_t)];
421 while (mask_block) {
422 // Get the lowest set bit in the mask block
423 size_t lowest_bit = mask_block & (-(ptrdiff_t)mask_block);
424 // Add it to the current subsignature if necessary
425 subsignature += (feature_block & lowest_bit) ? bit_index : 0;
426 // Reset the bit in the mask block
427 mask_block ^= lowest_bit;
428 // increment the bit index for the subsignature
429 bit_index <<= 1;
430 }
431 // Check the next feature block
432 ++feature_block_ptr;
433 }
434 return subsignature;
435}

437template<>
438inline LshStats LshTable<unsigned char>::getStats() const
439{
440 LshStats stats;
441 stats.bucket_size_mean_ = 0;
442 if ((buckets_speed_.empty()) && (buckets_space_.empty())) {
443 stats.n_buckets_ = 0;
444 stats.bucket_size_median_ = 0;
445 stats.bucket_size_min_ = 0;
446 stats.bucket_size_max_ = 0;
447 return stats;
448 }
449
450 if (!buckets_speed_.empty()) {
451 for (BucketsSpeed::const_iterator pbucket = buckets_speed_.begin(); pbucket != buckets_speed_.end(); ++pbucket) {
452 stats.bucket_sizes_.push_back((lsh::FeatureIndex)pbucket->size());
453 stats.bucket_size_mean_ += pbucket->size();
454 }
455 stats.bucket_size_mean_ /= buckets_speed_.size();
456 stats.n_buckets_ = buckets_speed_.size();
457 }
458 else {
459 for (BucketsSpace::const_iterator x = buckets_space_.begin(); x != buckets_space_.end(); ++x) {
460 stats.bucket_sizes_.push_back((lsh::FeatureIndex)x->second.size());
461 stats.bucket_size_mean_ += x->second.size();
462 }
463 stats.bucket_size_mean_ /= buckets_space_.size();
464 stats.n_buckets_ = buckets_space_.size();
465 }
466
467 std::sort(stats.bucket_sizes_.begin(), stats.bucket_sizes_.end());
468
469 // BOOST_FOREACH(int size, stats.bucket_sizes_)
470 // std::cout << size << " ";
471 // std::cout << std::endl;
472 stats.bucket_size_median_ = stats.bucket_sizes_[stats.bucket_sizes_.size() / 2];
473 stats.bucket_size_min_ = stats.bucket_sizes_.front();
474 stats.bucket_size_max_ = stats.bucket_sizes_.back();
475
476 // TODO compute mean and std
477 /*float mean, stddev;
478 stats.bucket_size_mean_ = mean;
479 stats.bucket_size_std_dev = stddev;*/
480
481 // Include a histogram of the buckets
482 unsigned int bin_start = 0;
483 unsigned int bin_end = 20;
484 bool is_new_bin = true;
485 for (std::vector<unsigned int>::iterator iterator = stats.bucket_sizes_.begin(), end = stats.bucket_sizes_.end(); iterator
486 != end; )
487 if (*iterator < bin_end) {
488 if (is_new_bin) {
489 stats.size_histogram_.push_back(std::vector<unsigned int>(3, 0));
490 stats.size_histogram_.back()[0] = bin_start;
491 stats.size_histogram_.back()[1] = bin_end - 1;
492 is_new_bin = false;
493 }
494 ++stats.size_histogram_.back()[2];
495 ++iterator;
496 }
497 else {
498 bin_start += 20;
499 bin_end += 20;
500 is_new_bin = true;
501 }
502
503 return stats;
504}
505
506// End the two namespaces
507}
