341static inline size_t fz_available(fz_context *ctx, fz_stream *stm, size_t max)
342{
343 size_t len = stm->wp - stm->rp;
344 int c = EOF;
345
346 if (len)
347 return len;
348 if (stm->eof)
349 return 0;
350
351 fz_try(ctx)
352 c = stm->next(ctx, stm, max);
353 fz_catch(ctx)
354 {
355 fz_rethrow_if(ctx, FZ_ERROR_TRYLATER);
356 fz_warn(ctx, "read error; treating as end of file");
357 stm->error = 1;
358 c = EOF;
359 }
360 if (c == EOF)
361 {
362 stm->eof = 1;
363 return 0;
364 }
365 stm->rp--;
366 return stm->wp - stm->rp;
367}

377static inline int fz_read_byte(fz_context *ctx, fz_stream *stm)
378{
379 int c = EOF;
380
381 if (stm->rp != stm->wp)
382 return *stm->rp++;
383 if (stm->eof)
384 return EOF;
385 fz_try(ctx)
386 c = stm->next(ctx, stm, 1);
387 fz_catch(ctx)
388 {
389 fz_rethrow_if(ctx, FZ_ERROR_TRYLATER);
390 fz_warn(ctx, "read error; treating as end of file");
391 stm->error = 1;
392 c = EOF;
393 }
394 if (c == EOF)
395 stm->eof = 1;
396 return c;
397}

406static inline int fz_peek_byte(fz_context *ctx, fz_stream *stm)
407{
408 int c = EOF;
409
410 if (stm->rp != stm->wp)
411 return *stm->rp;
412 if (stm->eof)
413 return EOF;
414
415 fz_try(ctx)
416 {
417 c = stm->next(ctx, stm, 1);
418 if (c != EOF)
419 stm->rp--;
420 }
421 fz_catch(ctx)
422 {
423 fz_rethrow_if(ctx, FZ_ERROR_TRYLATER);
424 fz_warn(ctx, "read error; treating as end of file");
425 stm->error = 1;
426 c = EOF;
427 }
428 if (c == EOF)
429 stm->eof = 1;
430 return c;
431}

452static inline int fz_is_eof(fz_context *ctx, fz_stream *stm)
453{
454 if (stm->rp == stm->wp)
455 {
456 if (stm->eof)
457 return 1;
458 return fz_peek_byte(ctx, stm) == EOF;
459 }
460 return 0;
461}

474static inline unsigned int fz_read_bits(fz_context *ctx, fz_stream *stm, int n)
475{
476 int x;
477
478 if (n <= stm->avail)
479 {
480 stm->avail -= n;
481 x = (stm->bits >> stm->avail) & ((1 << n) - 1);
482 }
483 else
484 {
485 x = stm->bits & ((1 << stm->avail) - 1);
486 n -= stm->avail;
487 stm->avail = 0;
488
489 while (n > 8)
490 {
491 x = (x << 8) | fz_read_byte(ctx, stm);
492 n -= 8;
493 }
494
495 if (n > 0)
496 {
497 stm->bits = fz_read_byte(ctx, stm);
498 stm->avail = 8 - n;
499 x = (x << n) | (stm->bits >> stm->avail);
500 }
501 }
502
503 return x;
504}

517static inline unsigned int fz_read_rbits(fz_context *ctx, fz_stream *stm, int n)
518{
519 int x;
520
521 if (n <= stm->avail)
522 {
523 x = stm->bits & ((1 << n) - 1);
524 stm->avail -= n;
525 stm->bits = stm->bits >> n;
526 }
527 else
528 {
529 unsigned int used = 0;
530
531 x = stm->bits & ((1 << stm->avail) - 1);
532 n -= stm->avail;
533 used = stm->avail;
534 stm->avail = 0;
535
536 while (n > 8)
537 {
538 x = (fz_read_byte(ctx, stm) << used) | x;
539 n -= 8;
540 used += 8;
541 }
542
543 if (n > 0)
544 {
545 stm->bits = fz_read_byte(ctx, stm);
546 x = ((stm->bits & ((1 << n) - 1)) << used) | x;
547 stm->avail = 8 - n;
548 stm->bits = stm->bits >> n;
549 }
550 }
551
552 return x;
553}
