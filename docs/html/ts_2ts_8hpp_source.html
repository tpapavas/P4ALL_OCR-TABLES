507class CV_EXPORTS BadArgTest : public BaseTest
508{
509public:
510 // constructor(s) and destructor
511 BadArgTest();
512 virtual ~BadArgTest();
513
514protected:
515 virtual int run_test_case( int expected_code, const string& descr );
516 virtual void run_func(void) = 0;
517 int test_case_idx;
518 int progress;
519 double t, freq;
520
521 template<class F>
522 int run_test_case( int expected_code, const string& _descr, F f)
523 {
524 double new_t = (double)cv::getTickCount(), dt;
525 if( test_case_idx < 0 )
526 {
527 test_case_idx = 0;
528 progress = 0;
529 dt = 0;
530 }
531 else
532 {
533 dt = (new_t - t)/(freq*1000);
534 t = new_t;
535 }
536 progress = update_progress(progress, test_case_idx, 0, dt);
537
538 int errcount = 0;
539 bool thrown = false;
540 const char* descr = _descr.c_str() ? _descr.c_str() : "";
541
542 try
543 {
544 f();
545 }
546 catch(const cv::Exception& e)
547 {
548 thrown = true;
549 if( e.code != expected_code )
550 {
551 ts->printf(TS::LOG, "%s (test case #%d): the error code %d is different from the expected %d\n",
552 descr, test_case_idx, e.code, expected_code);
553 errcount = 1;
554 }
555 }
556 catch(...)
557 {
558 thrown = true;
559 ts->printf(TS::LOG, "%s (test case #%d): unknown exception was thrown (the function has likely crashed)\n",
560 descr, test_case_idx);
561 errcount = 1;
562 }
563 if(!thrown)
564 {
565 ts->printf(TS::LOG, "%s (test case #%d): no expected exception was thrown\n",
566 descr, test_case_idx);
567 errcount = 1;
568 }
569 test_case_idx++;
570
571 return errcount;
572 }
573};
