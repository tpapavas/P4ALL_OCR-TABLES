229/*
230 Perform a depth first traversal of a tree.
231
232 Start at tree, looking for children in the array named
233 kid_name at each level.
234
235 The arrive callback is called when we arrive at a node (i.e.
236 before all the children are walked), and then the leave callback
237 is called as we leave it (after all the children have been
238 walked).
239
240 names and values are (matching) null terminated arrays of
241 names and values to be carried down the tree, to implement
242 inheritance. NULL is a permissible value.
243*/
244void pdf_walk_tree(fz_context *ctx, pdf_obj *tree, pdf_obj *kid_name,
245 void (*arrive)(fz_context *, pdf_obj *, void *, pdf_obj **),
246 void (*leave)(fz_context *, pdf_obj *, void *),
247 void *arg,
248 pdf_obj **names,
249 pdf_obj **values);

592/*
593 Regenerate any appearance streams that are out of date and check for
594 cases where a different appearance stream should be selected because of
595 state changes.
596
597 Note that a call to pdf_pass_event for one page may lead to changes on
598 any other, so an app should call pdf_update_annot for every annotation
599 it currently displays. Also it is important that the pdf_annot object
600 is the one used to last render the annotation. If instead the app were
601 to drop the page or annotations and reload them then a call to
602 pdf_update_annot would not reliably be able to report all changed
603 annotations.
604
605 Returns true if the annotation appearance has changed since the last time
606 pdf_update_annot was called or the annotation was first loaded.
607*/
608int pdf_update_annot(fz_context *ctx, pdf_annot *annot);
