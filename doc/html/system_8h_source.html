262/* Memory block alignment */
263
264/* Most architectures are happy with blocks being aligned to the size
265 * of void *'s. Some (notably sparc) are not.
266 *
267 * Some architectures (notably amd64) are happy for pointers to be 32bit
268 * aligned even on 64bit systems. By making use of this we can save lots
269 * of memory in data structures (notably the display list).
270 *
271 * We attempt to cope with these vagaries via the following definitions.
272 */
273
274/* All blocks allocated by mupdf's allocators are expected to be
275 * returned aligned to FZ_MEMORY_BLOCK_ALIGN_MOD. This is sizeof(void *)
276 * unless overwritten by a predefinition, or by a specific architecture
277 * being detected. */
278#ifndef FZ_MEMORY_BLOCK_ALIGN_MOD
279#if defined(sparc) || defined(__sparc) || defined(__sparc__)
280#define FZ_MEMORY_BLOCK_ALIGN_MOD 8
281#else
282#define FZ_MEMORY_BLOCK_ALIGN_MOD sizeof(void *)
283#endif
284#endif
285
286/* MuPDF will ensure that its use of pointers in packed structures
287 * (such as the display list) will be aligned to FZ_POINTER_ALIGN_MOD.
288 * This is the same as FZ_MEMORY_BLOCK_ALIGN_MOD unless overridden by
289 * a predefinition, or by a specific architecture being detected. */
290#ifndef FZ_POINTER_ALIGN_MOD
291#if defined(__amd64) || defined(__amd64__) || defined(__x86_64) || defined(__x86_64__)
292#define FZ_POINTER_ALIGN_MOD 4
293#else
294#define FZ_POINTER_ALIGN_MOD FZ_MEMORY_BLOCK_ALIGN_MOD
295#endif
296#endif

375static inline float my_sinf(float x)
376{
377 float x2, xn;
378 int i;
379 /* Map x into the -PI to PI range. We could do this using:
380 * x = fmodf(x, 2.0f * FZ_PI);
381 * but that's C99, and seems to misbehave with negative numbers
382 * on some platforms. */
383 x -= FZ_PI;
384 i = x / (2.0f * FZ_PI);
385 x -= i * 2.0f * FZ_PI;
386 if (x < 0.0f)
387 x += 2.0f * FZ_PI;
388 x -= FZ_PI;
389 if (x <= -FZ_PI / 2.0f)
390 x = -FZ_PI - x;
391 else if (x >= FZ_PI / 2.0f)
392 x = FZ_PI-x;
393 x2 = x * x;
394 xn = x * x2 / 6.0f;
395 x -= xn;
396 xn *= x2 / 20.0f;
397 x += xn;
398 xn *= x2 / 42.0f;
399 x -= xn;
400 xn *= x2 / 72.0f;
401 x += xn;
402 return x;
403}

405static inline float my_atan2f(float o, float a)
406{
407 int negate = 0, flip = 0, i;
408 float r, s;
409 if (o == 0.0f)
410 {
411 if (a > 0)
412 return 0.0f;
413 else
414 return FZ_PI;
415 }
416 if (o < 0)
417 o = -o, negate = 1;
418 if (a < 0)
419 a = -a, flip = 1;
420 if (o < a)
421 i = 65536.0f * o / a + 0.5f;
422 else
423 i = 65536.0f * a / o + 0.5f;
424 r = my_atan_table[i >> 8];
425 s = my_atan_table[(i >> 8) + 1];
426 r += (s - r) * (i & 255) / 256.0f;
427 if (o >= a)
428 r = FZ_PI / 2.0f - r;
429 if (flip)
430 r = FZ_PI - r;
431 if (negate)
432 r = -r;
433 return r;
434}
