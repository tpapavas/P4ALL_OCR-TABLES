220 {
221 assert(queries.cols == veclen());
222 assert(indices.rows >= queries.rows);
223 assert(dists.rows >= queries.rows);
224 assert(int(indices.cols) >= knn);
225 assert(int(dists.cols) >= knn);
226
227 KNNSimpleResultSet<DistanceType> resultSet(knn);
228 for (size_t i = 0; i < queries.rows; i++) {
229 resultSet.init(indices[i], dists[i]);
230 findNeighbors(resultSet, queries[i], params);

321 void computeBoundingBox(BoundingBox& bbox)
322 {
323 bbox.resize(dim_);
324 for (size_t i=0; i<dim_; ++i) {
325 bbox[i].low = (DistanceType)dataset_[0][i];
326 bbox[i].high = (DistanceType)dataset_[0][i];
327 }
328 for (size_t k=1; k<dataset_.rows; ++k) {
329 for (size_t i=0; i<dim_; ++i) {
330 if (dataset_[k][i]<bbox[i].low) bbox[i].low = (DistanceType)dataset_[k][i];
331 if (dataset_[k][i]>bbox[i].high) bbox[i].high = (DistanceType)dataset_[k][i];
332 }
333 }
334 }

346 NodePtr divideTree(int left, int right, BoundingBox& bbox)
347 {
348 NodePtr node = pool_.allocate<Node>(); // allocate memory
349
350 /* If too few exemplars remain, then make this a leaf node. */
351 if ( (right-left) <= leaf_max_size_) {
352 node->child1 = node->child2 = NULL; /* Mark as leaf node. */
353 node->left = left;
354 node->right = right;
355
356 // compute bounding-box of leaf points
357 for (size_t i=0; i<dim_; ++i) {
358 bbox[i].low = (DistanceType)dataset_[vind_[left]][i];
359 bbox[i].high = (DistanceType)dataset_[vind_[left]][i];
360 }
361 for (int k=left+1; k<right; ++k) {
362 for (size_t i=0; i<dim_; ++i) {
363 if (bbox[i].low>dataset_[vind_[k]][i]) bbox[i].low=(DistanceType)dataset_[vind_[k]][i];
364 if (bbox[i].high<dataset_[vind_[k]][i]) bbox[i].high=(DistanceType)dataset_[vind_[k]][i];
365 }
366 }
367 }
368 else {
369 int idx;
370 int cutfeat;
371 DistanceType cutval;
372 middleSplit_(&vind_[0]+left, right-left, idx, cutfeat, cutval, bbox);
373
374 node->divfeat = cutfeat;
375
376 BoundingBox left_bbox(bbox);
377 left_bbox[cutfeat].high = cutval;
378 node->child1 = divideTree(left, left+idx, left_bbox);
379
380 BoundingBox right_bbox(bbox);
381 right_bbox[cutfeat].low = cutval;
382 node->child2 = divideTree(left+idx, right, right_bbox);
383
384 node->divlow = left_bbox[cutfeat].high;
385 node->divhigh = right_bbox[cutfeat].low;
386
387 for (size_t i=0; i<dim_; ++i) {
388 bbox[i].low = std::min(left_bbox[i].low, right_bbox[i].low);
389 bbox[i].high = std::max(left_bbox[i].high, right_bbox[i].high);
390 }
391 }
392
393 return node;
394 }

396 void computeMinMax(int* ind, int count, int dim, ElementType& min_elem, ElementType& max_elem)
397 {
398 min_elem = dataset_[ind[0]][dim];
399 max_elem = dataset_[ind[0]][dim];
400 for (int i=1; i<count; ++i) {
401 ElementType val = dataset_[ind[i]][dim];
402 if (val<min_elem) min_elem = val;
403 if (val>max_elem) max_elem = val;
404 }

407 void middleSplit(int* ind, int count, int& index, int& cutfeat, DistanceType& cutval, const BoundingBox& bbox)
408 {
409 // find the largest span from the approximate bounding box
410 ElementType max_span = bbox[0].high-bbox[0].low;
411 cutfeat = 0;
412 cutval = (bbox[0].high+bbox[0].low)/2;
413 for (size_t i=1; i<dim_; ++i) {
414 ElementType span = bbox[i].high-bbox[i].low;
415 if (span>max_span) {
416 max_span = span;
417 cutfeat = i;
418 cutval = (bbox[i].high+bbox[i].low)/2;
419 }
420 }
421
422 // compute exact span on the found dimension
423 ElementType min_elem, max_elem;
424 computeMinMax(ind, count, cutfeat, min_elem, max_elem);
425 cutval = (min_elem+max_elem)/2;
426 max_span = max_elem - min_elem;
427
428 // check if a dimension of a largest span exists
429 size_t k = cutfeat;
430 for (size_t i=0; i<dim_; ++i) {
431 if (i==k) continue;
432 ElementType span = bbox[i].high-bbox[i].low;
433 if (span>max_span) {
434 computeMinMax(ind, count, i, min_elem, max_elem);
435 span = max_elem - min_elem;
436 if (span>max_span) {
437 max_span = span;
438 cutfeat = i;
439 cutval = (min_elem+max_elem)/2;
440 }
441 }
442 }
443 int lim1, lim2;
444 planeSplit(ind, count, cutfeat, cutval, lim1, lim2);
445
446 if (lim1>count/2) index = lim1;
447 else if (lim2<count/2) index = lim2;
448 else index = count/2;
449 }

452 void middleSplit_(int* ind, int count, int& index, int& cutfeat, DistanceType& cutval, const BoundingBox& bbox)
453 {
454 const float EPS=0.00001f;
455 DistanceType max_span = bbox[0].high-bbox[0].low;
456 for (size_t i=1; i<dim_; ++i) {
457 DistanceType span = bbox[i].high-bbox[i].low;
458 if (span>max_span) {
459 max_span = span;
460 }
461 }
462 DistanceType max_spread = -1;
463 cutfeat = 0;
464 for (size_t i=0; i<dim_; ++i) {
465 DistanceType span = bbox[i].high-bbox[i].low;
466 if (span>(DistanceType)((1-EPS)*max_span)) {
467 ElementType min_elem, max_elem;
468 computeMinMax(ind, count, cutfeat, min_elem, max_elem);
469 DistanceType spread = (DistanceType)(max_elem-min_elem);
470 if (spread>max_spread) {
471 cutfeat = (int)i;
472 max_spread = spread;
473 }
474 }
475 }
476 // split in the middle
477 DistanceType split_val = (bbox[cutfeat].low+bbox[cutfeat].high)/2;
478 ElementType min_elem, max_elem;
479 computeMinMax(ind, count, cutfeat, min_elem, max_elem);
480
481 if (split_val<min_elem) cutval = (DistanceType)min_elem;
482 else if (split_val>max_elem) cutval = (DistanceType)max_elem;
483 else cutval = split_val;
484
485 int lim1, lim2;
486 planeSplit(ind, count, cutfeat, cutval, lim1, lim2);
487
488 if (lim1>count/2) index = lim1;
489 else if (lim2<count/2) index = lim2;
490 else index = count/2;
491 }

503 void planeSplit(int* ind, int count, int cutfeat, DistanceType cutval, int& lim1, int& lim2)
504 {
505 /* Move vector indices for left subtree to front of list. */
506 int left = 0;
507 int right = count-1;
508 for (;; ) {
509 while (left<=right && dataset_[ind[left]][cutfeat]<cutval) ++left;
510 while (left<=right && dataset_[ind[right]][cutfeat]>=cutval) --right;
511 if (left>right) break;
512 std::swap(ind[left], ind[right]); ++left; --right;
513 }
514 /* If either list is empty, it means that all remaining features
515 * are identical. Split in the middle to maintain a balanced tree.
516 */
517 lim1 = left;
518 right = count-1;
519 for (;; ) {
520 while (left<=right && dataset_[ind[left]][cutfeat]<=cutval) ++left;
521 while (left<=right && dataset_[ind[right]][cutfeat]>cutval) --right;
522 if (left>right) break;
523 std::swap(ind[left], ind[right]); ++left; --right;
524 }
525 lim2 = left;
526 }

528 DistanceType computeInitialDistances(const ElementType* vec, std::vector<DistanceType>& dists)
529 {
530 DistanceType distsq = 0.0;
531
532 for (size_t i = 0; i < dim_; ++i) {
533 if (vec[i] < root_bbox_[i].low) {
534 dists[i] = distance_.accum_dist(vec[i], root_bbox_[i].low, (int)i);
535 distsq += dists[i];
536 }
537 if (vec[i] > root_bbox_[i].high) {
538 dists[i] = distance_.accum_dist(vec[i], root_bbox_[i].high, (int)i);
539 distsq += dists[i];
540 }
541 }
542
543 return distsq;
544 }

549 void searchLevel(ResultSet<DistanceType>& result_set, const ElementType* vec, const NodePtr node, DistanceType mindistsq,
550 std::vector<DistanceType>& dists, const float epsError)
551 {
552 /* If this is a leaf node, then do check and return. */
553 if ((node->child1 == NULL)&&(node->child2 == NULL)) {
554 DistanceType worst_dist = result_set.worstDist();
555 for (int i=node->left; i<node->right; ++i) {
556 int index = reorder_ ? i : vind_[i];
557 DistanceType dist = distance_(vec, data_[index], dim_, worst_dist);
558 if (dist<worst_dist) {
559 result_set.addPoint(dist,vind_[i]);
560 }
561 }
562 return;
563 }
564
565 /* Which child branch should be taken first? */
566 int idx = node->divfeat;
567 ElementType val = vec[idx];
568 DistanceType diff1 = val - node->divlow;
569 DistanceType diff2 = val - node->divhigh;
570
571 NodePtr bestChild;
572 NodePtr otherChild;
573 DistanceType cut_dist;
574 if ((diff1+diff2)<0) {
575 bestChild = node->child1;
576 otherChild = node->child2;
577 cut_dist = distance_.accum_dist(val, node->divhigh, idx);
578 }
579 else {
580 bestChild = node->child2;
581 otherChild = node->child1;
582 cut_dist = distance_.accum_dist( val, node->divlow, idx);
583 }
584
585 /* Call recursively to search next level down. */
586 searchLevel(result_set, vec, bestChild, mindistsq, dists, epsError);
587
588 DistanceType dst = dists[idx];
589 mindistsq = mindistsq + cut_dist - dst;
590 dists[idx] = cut_dist;
591 if (mindistsq*epsError<=result_set.worstDist()) {
592 searchLevel(result_set, vec, otherChild, mindistsq, dists, epsError);
593 }
594 dists[idx] = dst;
595 }
