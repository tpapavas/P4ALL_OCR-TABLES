56 The various types of document events

59/*
60 details of an alert event. In response the app should
61 display an alert dialog with the buttons specified by "button_type_group".
62 If "check_box_message" is non-NULL, a checkbox should be displayed in
63 the lower-left corned along with the message.
64
65 "finally_checked" and "button_pressed" should be set by the app
66 before returning from the callback. "finally_checked" need be set
67 only if "check_box_message" is non-NULL.
68*/
69typedef struct
70{
71 const char *message;
72 int icon_type;
73 int button_group_type;
74 const char *title;
75 const char *check_box_message;
76 int initially_checked;
77 int finally_checked;
78 int button_pressed;
79} pdf_alert_event;
