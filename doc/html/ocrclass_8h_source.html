38/**********************************************************************
39 * EANYCODE_CHAR
40 * Description of a single character. The character code is defined by
41 * the character set of the current font.
42 * Output text is sent as an array of these structures.
43 * Spaces and line endings in the output are represented in the
44 * structures of the surrounding characters. They are not directly
45 * represented as characters.
46 * The first character in a word has a positive value of blanks.
47 * Missing information should be set to the defaults in the comments.
48 * If word bounds are known, but not character bounds, then the top and
49 * bottom of each character should be those of the word. The left of the
50 * first and right of the last char in each word should be set. All other
51 * lefts and rights should be set to -1.
52 * If set, the values of right and bottom are left+width and top+height.
53 * Most of the members come directly from the parameters to ocr_append_char.
54 * The formatting member uses the enhancement parameter and combines the
55 * line direction stuff into the top 3 bits.
56 * The coding is 0=RL char, 1=LR char, 2=DR NL, 3=UL NL, 4=DR Para,
57 * 5=UL Para, 6=TB char, 7=BT char. API users do not need to know what
58 * the coding is, only that it is backwards compatible with the previous
59 * version.
60 **********************************************************************/
61
62typedef struct { /*single character */
63 // It should be noted that the format for char_code for version 2.0 and beyond
64 // is UTF8 which means that ASCII characters will come out as one structure
65 // but other characters will be returned in two or more instances of this
66 // structure with a single byte of the UTF8 code in each, but each will have
67 // the same bounding box. Programs which want to handle languagues with
68 // different characters sets will need to handle extended characters
69 // appropriately, but *all* code needs to be prepared to receive UTF8 coded
70 // characters for characters such as bullet and fancy quotes.
71 uint16_t char_code; /*character itself */
72 int16_t left; /*of char (-1) */
73 int16_t right; /*of char (-1) */
74 int16_t top; /*of char (-1) */
75 int16_t bottom; /*of char (-1) */
76 int16_t font_index; /*what font (0) */
77 uint8_t confidence; /*0=perfect, 100=reject (0/100) */
78 uint8_t point_size; /*of char, 72=i inch, (10) */
79 int8_t blanks; /*no of spaces before this char (1) */
80 uint8_t formatting; /*char formatting (0) */
81} EANYCODE_CHAR; /*single character */

83/**********************************************************************
84 * ETEXT_DESC
85 * Description of the output of the OCR engine.
86 * This structure is used as both a progress monitor and the final
87 * output header, since it needs to be a valid progress monitor while
88 * the OCR engine is storing its output to shared memory.
89 * During progress, all the buffer info is -1.
90 * Progress starts at 0 and increases to 100 during OCR. No other constraint.
91 * Additionally the progress callback contains the bounding box of the word that
92 * is currently being processed.
93 * Every progress callback, the OCR engine must set ocr_alive to 1.
94 * The HP side will set ocr_alive to 0. Repeated failure to reset
95 * to 1 indicates that the OCR engine is dead.
96 * If the cancel function is not null then it is called with the number of
97 * user words found. If it returns true then operation is cancelled.
98 **********************************************************************/
99class ETEXT_DESC;

133 if (deadline_msecs > 0) {

140 // Returns false if we've not passed the end_time, or have not set a deadline.
141 bool deadline_exceeded() const {
142 if (end_time.tv_sec == 0 && end_time.tv_usec == 0)
143 return false;
144 auto chrono_now = std::chrono::steady_clock::now();
145 struct timeval now;
146 timePointToTimeval(chrono_now, &now);
147 return (now.tv_sec > end_time.tv_sec ||
148 (now.tv_sec == end_time.tv_sec && now.tv_usec > end_time.tv_usec));
149 }
150
151 private:
152 static void timePointToTimeval(
153 std::chrono::steady_clock::time_point chrono_point, struct timeval* tv) {
154 auto millisecs = std::chrono::duration_cast<std::chrono::milliseconds>(
155 chrono_point.time_since_epoch());
156 tv->tv_sec = millisecs.count() / 1000;
157 tv->tv_usec = (millisecs.count() % 1000) * 1000;
158 }
159
160 static bool default_progress_func(ETEXT_DESC* ths, int left, int right,
161 int top, int bottom) {
162 if (ths->progress_callback != nullptr) {
163 return (*(ths->progress_callback))(ths->progress, left, right, top,
164 bottom);
165 }
166 return true;
167 }
168};
