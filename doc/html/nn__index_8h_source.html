109 virtual int radiusSearch(const Matrix<ElementType>& query, Matrix<int>& indices, Matrix<DistanceType>& dists, float radius, const SearchParams& params)
110 {
111 if (query.rows != 1) {
112 fprintf(stderr, "I can only search one feature at a time for range search\n");
113 return -1;
114 }
115 assert(query.cols == veclen());
116 assert(indices.cols == dists.cols);
117
118 int n = 0;
119 int* indices_ptr = NULL;
120 DistanceType* dists_ptr = NULL;
121 if (indices.cols > 0) {
122 n = (int)indices.cols;
123 indices_ptr = indices[0];
124 dists_ptr = dists[0];
125 }
126
127 RadiusUniqueResultSet<DistanceType> resultSet((DistanceType)radius);
128 resultSet.clear();
129 findNeighbors(resultSet, query[0], params);
130 if (n>0) {
131 if (get_param(params,"sorted",true)) resultSet.sortAndCopy(indices_ptr, dists_ptr, n);
132 else resultSet.copy(indices_ptr, dists_ptr, n);
133 }
134
135 return (int)resultSet.size();
136 }
